Actor BdChainsaw : BdLiteWeapon replaces Chainsaw
{
	weapon.selectionorder 220
	Weapon.SlotNumber 1
	Weapon.SlotPriority 1
	inventory.pickupsound "weapons/chainsaw/draw"
    Inventory.PickupMessage "$GOTCHAINSAW"
    Obituary "$OB_MPCHAINSAW"
    Tag "$TAG_CHAINSAW"
	Weapon.UpSound "none"
	weapon.readysound "weapons/chainsaw/idle"
	+WEAPON.AXEBLOOD
	+WEAPON.MELEEWEAPON
states
	{
	Spawn:
		CSAW A -1
	Deselect:
		TNT1 A 0 A_StopsoundEx("Weapon")
		TNT1 A 0 A_Lower
		SAWS A 1 A_Lower
		wait
	Ready:
		SAWY ABCDE 1
	    SAWS A 6
        TNT1 A 0 A_PlaySound("DSSAWZIP", 1)
		TNT1 A 0 A_playsound("weapons/chainsaw/idle", 2)
        SAWS BCDEFEDCB 1
		SAWS A 1
		SAWS A 6 A_playsound("weapons/chainsaw/draw", 1)
		SAWS A 2 A_PlaySound("DSSAWUP", 2)
		SAWS AGAGAGAG 2
	ReadyLoop:
		TNT1 A 0 A_playsound("weapons/chainsaw/idle")
		SAWG ABAB 2 A_weaponready(WRF_ALLOWZOOM)
		goto ReadyLoop
	Fire:
	    TNT1 A 0
		TNT1 A 0 A_playsound("weapons/chainsaw/start")
		SAWG ABCD 1
	Hold:
		TNT1 A 0 A_PlaySoundEx("weapons/chainsaw/loop","Weapon",0)
		SAWF A 1 A_Saw("weapons/chainsaw/loop", "CSAWCUT", 5, "SSawPuff")
		SAWF B 1
		SAWF C 1 A_Saw("", "", 5, "SSawPuff")
		SAWF D 1
		
		TNT1 A 0 A_Refire
		TNT1 A 0 A_StopsoundEx("Weapon")
		TNT1 A 0 A_Playsound("weapons/chainsaw/stop")
		SAWG DC 1
		goto ReadyLoop
	}
}
